#include<stdio.h>
 
int main()
{
 int Size, i, a[10];
 int Even_Count = 0, Odd_Count = 0;
 
 printf("\n Please Enter the Size of an Array :  ");
 scanf("%d", &Size);
 
 printf("\nPlease Enter the Array Elements\n");
 for(i = 0; i < Size; i++)
 {
   scanf("%d", &a[i]);
 }
  
 for(i = 0; i < Size; i ++)
 {
   if(a[i] % 2 == 0)
   {
     Even_Count++;
   }
   else
   {
     Odd_Count++;
   }
 }
  
 printf("\nEven Numbers= %d ", Even_Count);
 printf("\nOdd Numbers= %d ", Odd_Count);
 return 0;
}
